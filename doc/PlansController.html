    def create
        @plan = Plan.new(params[:plan])
        @plan.plan_id = SecureRandom.urlsafe_base64(20)
        if @plan.save
            redirect_to plan_url(@plan.plan_id), notice: "Plan made!"
            session[:plan_id] = @plan.plan_id
#            UserMailer.url_mail(@user).deliver
        else
            render 'new'
        end
    end
